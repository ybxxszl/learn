常用命令
	1、管理员权限
		su命令
			重装系统后需要设置root密码
				sudo passwd root
	2、更新缓存
		apt-get update
	3、更新软件
		apt-get upgrade
	4、自动解决依赖
		apt-get install -f
	5、删除没有使用的依赖
		apt-get install autoremove
	6、安装.deb文件
		dpkg -i sougoupinyin.deb
	7、运行.sh文件
		./jmeter.sh
	8、运行.jar文件
		java -jar restclient.jar


创建快捷方式	/usr/share/applications/Eclipse.desktop
[Desktop Entry]
Encoding=UTF-8
Name=Eclipse
Comment=Eclipse IDE
Exec=/home/ybxxszl/application/Eclipse/eclipse/eclipse
Icon=/home/ybxxszl/application/Eclipse/eclipse/icon.xpm
Terminal=false
StartupNotify=true
Type=Application
Categories=Application;Development;
